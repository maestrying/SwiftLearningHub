```swift
let doubled = moreInts.map { $0 * 2 }  // –£–¥–≤–∞–∏–≤–∞–µ—Ç –∫–∞–∂–¥—ã–π —ç–ª–µ–º–µ–Ω—Ç
```

## –ò–Ω—Å—Ç—Ä—É–∫—Ü–∏—è if
–ö–∞–∂–¥—ã–π –æ–ø–µ—Ä–∞—Ç–æ—Ä —Å—Ä–∞–≤–Ω–µ–Ω–∏—è –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç –∑–Ω–∞—á–µ–Ω–∏–µ —Ç–∏–ø–∞ Bool, —É–∫–∞–∑—ã–≤–∞—é—â–µ–µ, —è–≤–ª—è–µ—Ç—Å—è –ª–∏ –≤—ã—Ä–∞–∂–µ–Ω–∏–µ –∏—Å—Ç–∏–Ω–Ω—ã–º:
```swift
let name = "world"

if name == "world" {
    print("hello, world")
} else if name == "Max" {
    print("hello, Max")
} else {
    print("–ú–Ω–µ –∂–∞–ª—å, \(name), –Ω–æ —è —Ç–µ–±—è –Ω–µ —É–∑–Ω–∞—é")
}
// –Ω–∞–ø–µ—á–∞—Ç–∞–µ—Ç "hello, world"
```

–¢–∞–∫ –∂–µ –º–æ–∂–Ω–æ —Å—Ä–∞–≤–Ω–∏–≤–∞—Ç—å –∫–æ—Ä—Ç–µ–∂–∏, –∫–æ—Ç–æ—Ä—ã–µ –∏–º–µ—é—Ç –æ–¥–Ω–æ –∏ —Ç–æ –∂–µ –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –∑–Ω–∞—á–µ–Ω–∏–π, –∫–æ—Ç–æ—Ä—ã–µ, –≤ —Å–≤–æ—é –æ—á–µ—Ä–µ–¥—å, –¥–æ–ª–∂–Ω—ã –±—ã—Ç—å —Å—Ä–∞–≤–Ω–∏–≤–∞–µ–º—ã–º–∏, —á—Ç–æ –æ–∑–Ω–∞—á–∞–µ—Ç, —á—Ç–æ –∫–æ—Ä—Ç–µ–∂ —Ç–∏–ø–∞ (`Int`, `String`) –º–æ–∂–µ—Ç –±—ã—Ç—å —Å—Ä–∞–≤–Ω–µ–Ω —Å –∫–æ—Ä—Ç–µ–∂–µ–º —Ç–∞–∫–æ–≥–æ –∂–µ —Ç–∏–ø–∞.

–ö–æ—Ä—Ç–µ–∂–∏ —Å—Ä–∞–≤–Ω–∏–≤–∞—é—Ç—Å—è —Å–ª–µ–≤–∞ –Ω–∞–ø—Ä–∞–≤–æ, –ø–æ –æ–¥–Ω–æ–º—É –∑–Ω–∞—á–µ–Ω–∏—é –∑–∞ —Ä–∞–∑ –¥–æ —Ç–µ—Ö –ø–æ—Ä, –ø–æ–∫–∞ –æ–ø–µ—Ä–∞—Ü–∏—è —Å—Ä–∞–≤–Ω–µ–Ω–∏—è –Ω–µ –Ω–∞–π–¥–µ—Ç –æ—Ç–ª–∏—á–∏—è –º–µ–∂–¥—É –∑–Ω–∞—á–µ–Ω–∏—è–º–∏. –ï—Å–ª–∏ –≤—Å–µ –∑–Ω–∞—á–µ–Ω–∏—è –∫–æ—Ä—Ç–µ–∂–µ–π –ø–æ–ø–∞—Ä–Ω–æ —Ä–∞–≤–Ω—ã, —Ç–æ –∏ –∫–æ—Ä—Ç–µ–∂–∏ —Ç–∞–∫ –∂–µ —Å—á–∏—Ç–∞—é—Ç—Å—è —Ä–∞–≤–Ω—ã–º–∏
```swift
(1, "zebra") < (2, "apple")   // true, –ø–æ—Ç–æ–º—É —á—Ç–æ 1 –º–µ–Ω—å—à–µ 2, "zebra" –∏ "apple" –Ω–µ —Å—Ä–∞–≤–Ω–∏–≤–∞—é—Ç—Å—è
```

## –ò–Ω—Å—Ç—Ä—É–∫—Ü–∏—è switch
`Switch` –ø–æ–¥—Ä–∞–∑—É–º–µ–≤–∞–µ—Ç –Ω–∞–ª–∏—á–∏–µ –∫–∞–∫–æ–≥–æ-—Ç–æ –∑–Ω–∞—á–µ–Ω–∏—è, –∫–æ—Ç–æ—Ä–æ–µ —Å—Ä–∞–≤–Ω–∏–≤–∞–µ—Ç—Å—è —Å –Ω–µ—Å–∫–æ–ª—å–∫–∏–º–∏ –≤–æ–∑–º–æ–∂–Ω—ã–º–∏ —à–∞–±–ª–æ–Ω–∞–º–∏. –ü–æ—Å–ª–µ —Ç–æ–≥–æ –∫–∞–∫ –∑–Ω–∞—á–µ–Ω–∏–µ —Å–æ–≤–ø–∞–ª–æ —Å –∫–∞–∫–∏–º-–ª–∏–±–æ —à–∞–±–ª–æ–Ω–æ–º, –≤—ã–ø–æ–ª–Ω—è–µ—Ç—Å—è –∫–æ–¥, —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤—É—é—â–∏–π –æ—Ç–≤–µ—Ç–≤–ª–µ–Ω–∏—é —ç—Ç–æ–≥–æ —à–∞–±–ª–æ–Ω–∞, –∏ –±–æ–ª—å—à–µ —Å—Ä–∞–≤–Ω–µ–Ω–∏–π —É–∂–µ –Ω–µ –ø—Ä–æ–∏—Å—Ö–æ–¥–∏—Ç. `Switch` –ø—Ä–µ–¥—Å—Ç–∞–≤–ª—è–µ—Ç —Å–æ–±–æ–π –∞–ª—å—Ç–µ—Ä–Ω–∞—Ç–∏–≤—É –∏–Ω—Å—Ç—Ä—É–∫—Ü–∏–∏ `if`, –æ—Ç–≤–µ—á–∞—é—â–µ–π –Ω–µ—Å–∫–æ–ª—å–∫–∏–º –ø–æ—Ç–µ–Ω—Ü–∏–∞–ª—å–Ω—ã–º –∑–Ω–∞—á–µ–Ω–∏—è–º:
```swift
let fruit = "apple"

switch fruit {

    print("–≠—Ç–æ —è–±–ª–æ–∫–æ.")
case "banana":
    print("–≠—Ç–æ –±–∞–Ω–∞–Ω.")
case "orange":
    print("–≠—Ç–æ –∞–ø–µ–ª—å—Å–∏–Ω.")

    print("–ù–µ–∏–∑–≤–µ—Å—Ç–Ω—ã–π —Ñ—Ä—É–∫—Ç.")
}
// –≠—Ç–æ —è–±–ª–æ–∫–æ.
```

## –ö–æ—Ä—Ç–µ–∂–∏ (Tuples)
–ö–æ—Ä—Ç–µ–∂–∏ –∏–ª–∏ Tuples –ø—Ä–µ–¥—Å—Ç–∞–≤–ª—è—é—Ç –Ω–∞–±–æ—Ä –∑–Ω–∞—á–µ–Ω–∏–π, –∫–æ—Ç–æ—Ä—ã–µ —Ä–∞—Å—Å–º–∞—Ç—Ä–∏–≤–∞—é—Ç—Å—è –∫–∞–∫ –æ–¥–∏–Ω –æ–±—ä–µ–∫—Ç:

–í –¥–∞–Ω–Ω–æ–º –ø—Ä–∏–º–µ—Ä–µ —Ç–∏–ø –¥–∞–Ω–Ω—ã—Ö –≤ –∫–æ—Ä—Ç–µ–∂–∞—Ö –Ω–µ —É–∫–∞–∑–∞–Ω –∏ –≤—ã–≤–æ–¥–∏—Ç—Å—è –Ω–µ—è–≤–Ω–æ, –Ω–æ –º—ã –º–æ–∂–µ–º –ø—Ä–∏ –æ–ø—Ä–µ–¥–µ–ª–µ–Ω–∏–∏ –∫–æ—Ä—Ç–µ–∂–∞ —Ç–∞–∫–∂–µ –æ–±—ä—è–≤–∏—Ç—å –∏ —Ç–∏–ø—ã –¥–∞–Ω–Ω—ã—Ö –µ–≥–æ –∑–Ω–∞—á–µ–Ω–∏–π:
